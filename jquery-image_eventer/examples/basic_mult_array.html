Multiple arrays

Loading multiple sets of images in a particular order is easy. Pass 'files' (or whatever makes sense to you) as the second argument in your handler fn. With this second argument you get back EXACTLY what was passed in your collections array. This will allow you to manipulate the DOM as you wish at that point.



Our [0] point in our collections array will be loaded first. After that our [1] point in our collections array will be loaded.

var $body = $('body'),
    accordion = 'imgs/holidays/accordion.jpg',
    cycle = 'imgs/holidays/cycle.jpg',
    gothing = 'imgs/holidays/gothing.jpg',
    smoking1 = 'imgs/holidays/smoking1.jpg',
    smoking2 = 'imgs/holidays/smoking2.jpg',
    stalking = 'imgs/holidays/stalking.jpg',
    treehats = 'imgs/holidays/treehats.jpg',
    $left = $('.left'),
    $right = $('.right');

$body
  .bind('image_eventer::imagesLoadComplete', function(e, files){
    var fLen = files.length,
        liBuffer = [],
        $target;

  if($left.find('li').length === 0){
    $target = $left;
  } else {
    $target = $right;
  }

    for(var i=0; i<fLen; i++){
      liBuffer.push('<li style="height: 500px"><img width=300 src="'+files[i]+'" alt="holiday rad-ness" /></li>');
    }

    $target.append(liBuffer.join(''));

  })
  .image_eventer({
    collections: [
      [accordion, cycle, gothing, smoking1],
      [smoking2, stalking, treehats]
    ]
  });
