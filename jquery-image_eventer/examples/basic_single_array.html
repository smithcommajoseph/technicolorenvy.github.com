Single array

Supposing we have an unordered list with 4 li elements. If each of those elements have an unique ID, we can 'preload' our images then append them into their corresponding li like so.



If you are seeing images above, our 'image_eventer::imagesLoadComplete' successfully fired. Since we were listening to that event, we were able to append our images into their proper place.

var $body = $('body'),
  f1 = 'imgs/waters/faces1.jpg',
  f2 = 'imgs/waters/faces2.jpg',
  fly = 'imgs/waters/flying.jpg',
  f3 = 'imgs/waters/faces3.jpg',
  $f1 = $('#f1'),
  $f2 = $('#f2'),
  $fly = $('#fly'),
  $f3 = $('#f3');

$body
  .bind('image_eventer::imagesLoadComplete', function(e, files){
    $f1.html('<img width=300 src="'+f1+'" alt="watersports" />');
    $f2.html('<img width=300 src="'+f2+'" alt="watersports" />');
    $fly.html('<img width=300 src="'+fly+'" alt="watersports" />');
    $f3.html('<img width=300 src="'+f3+'" alt="watersports" />');
  })
  .image_eventer([ f1, f2, fly, f3 ]);
