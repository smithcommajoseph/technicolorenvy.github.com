Mutliple jQ parents

Loading multiple sets of images in order that already exist in the DOM



Our [0] point in our collections array will be loaded first. After that our [1] point in our collections array will be loaded.

var $body = $('body'),
    $left = $('.left'),
    $right = $('.right');

$body
  .bind('image_eventer::imagesLoadComplete', function(e){
    var $target;

    if($left.hasClass('hidden')){
      $target = $left;
    } else {
      $target = $right;
    }

    $target.removeClass('hidden').fadeIn();

  })
  .image_eventer({
    collections: [
      $left,
      $right
    ]
  });
