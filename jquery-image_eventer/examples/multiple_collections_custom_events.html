Multiple collections w/ custom events

Loading multiple sets of images in order. Some exist in the DOM, some do not.



Our [0] point in our collections array will be loaded first. After that our [1] point in our collections array will be loaded.

Each collection will fire it's own event.

var $body = $('body'),
    bighead = 'imgs/awkward_fam_photos/bighead.jpg',
    bookends = 'imgs/awkward_fam_photos/bookends.jpg',
    bufftans = 'imgs/awkward_fam_photos/bufftans.jpg',
    centaur = 'imgs/awkward_fam_photos/centaur.jpg',
    headpile = 'imgs/awkward_fam_photos/headpile.jpg',
    hearmenow = 'imgs/awkward_fam_photos/hearmenow.jpg',
    $left = $('.left'),
    $right = $('.right');

$body
  .bind({
    'familyCrude': function(e, files){
      var fLen = files.length,
          i, frags = [];

      for(i=0; i<files.length; i++){
        frags.push('<li><img width=300 src="'+files[i]+'" alt="awkward family photos" /></li>');
      }

      $left.html(frags.join('')).fadeIn().removeClass('hidden');

    },
    'familyFeud': function(e){
      $right.fadeIn().removeClass('hidden');
    }

})
  .image_eventer({
    collections: [
      {
        files: [bighead, bookends, bufftans, centaur, headpile, hearmenow],
        completedEvent: 'familyCrude'
      },
      {
        files: $right,
        completedEvent: 'familyFeud'
      }
    ]
  });
